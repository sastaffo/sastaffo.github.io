# sets up prefixes

@prefix rdf: .

@prefix rdfs: .

@prefix owl: .

@prefix foaf: .

@prefix my: .

# sets up classes

my:Person rdfs:subClassOf foaf:Person .

my:Student rdfs:subClassOf my:Person .

# sets up predicates

my:isParentOf a rdf:predicate ;

rdfs:isDefinedBy ;

rdfs:label "isParentOf" ;

rdfs:comment "Subject is a parent of the Object." ;

rdfs:domain my:Person ; # subject must be type my:person

rdfs:range my:Person . # object must be type my:person

my:isChildOf a rdf:predicate ;

rdfs:isDefinedBy ;

rdfs:label "isChildOf" ;

rdfs:comment "Subject is a child of the Object." ;

rdfs:domain my:Person ;

rdfs:range my:Person .

my:isSiblingOf a rdf:predicate ;

rdfs:isDefinedBy ;

rdfs:label "isSiblingOf" ;

rdfs:comment "Subject is a sibling of the Object." ;

rdfs:domain my:Person ;

rdfs:range my:Person .

my:firstName a rdf:predicate ;

rdfs:isDefinedBy ;

rdfs:label "firstName" ;

rdfs:comment "Subject has first name Object, as String." ;

rdfs:domain my:Person ;

rdfs:range rdfs:Literal .

my:lastName a rdf:predicate ;

rdfs:isDefinedBy ;

rdfs:label "lastName" ;

rdfs:comment "Subject has last name Object, as String." ;

rdfs:domain my:Person ;

rdfs:range rdfs:Literal .

my:dateOfBirth a rdf:predicate ;

rdfs:isDefinedBy ;

rdfs:label "dateOfBirth" ;

rdfs:comment "Subject was born Object, as Date." ;

rdfs:domain my:Person ;

rdfs:range rdfs:Literal .
